correlation  <- function(x,y,z){
    #   x_mean   sum0  sum2  
    #     y_mean    sum1   
    #   r   sum3,sum4,sum5,x_mean, y_mean  
    #    sum0  sum5  0 . 
    sum0  <- 0    #       x   
    sum1  <- 0    #       x   
    sum2  <- 0    #   
    sum3  <- 0    # x y 
    sum4  <- 0    # x 
    sum5  <- 0    # y  
    
    #   sum0,sum1,sum2  
    for (i in 1:length(x)){
        sum0 = sum0 + z[i]*x[i]
        sum1 = sum1 + z[i]*y[i]
        sum2 = sum2 + z[i]
    }
    
    #        
    x_mean  <- sum0/sum2
    y_mean  <- sum1/sum2
    
    #   sum3,sum4,sum5 
    for (i in 1:length(x)){
        sum3 = sum3 + z[i]*((x[i]-x_mean)*(y[i]-y_mean))
        sum4 = sum4 + z[i]*((x[i]-x_mean)^2)
        sum5 = sum5 + z[i]*((y[i]-y_mean)^2)
        
    }
    
    #  r 
    r  <- sum3/sqrt(sum4*sum5)
    return (r)
}
